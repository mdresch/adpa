```sql
-- Resource Breakdown Structure (Microsoft PPM equivalent)
CREATE TABLE portfolio_resource_breakdown_structure (
  id UUID PRIMARY KEY,
  organization_id UUID,
  
  -- RBS Hierarchy (Microsoft PPM format)
  rbs_code VARCHAR(50),           -- e.g., "ENG.DEV.SENIOR"
  rbs_level INTEGER,              -- 1 = top, 2 = department, 3 = role
  parent_rbs_id UUID REFERENCES portfolio_resource_breakdown_structure(id),
  
  -- Classification
  rbs_name VARCHAR(255),          -- "Senior Developer"
  rbs_category VARCHAR(100),      -- department, role, skill, location
  
  -- Capacity
  total_capacity DECIMAL(10,2),   -- FTE or hours
  capacity_unit VARCHAR(50),      -- fte, hours, days
  
  -- Organizational
  department VARCHAR(255),
  division VARCHAR(255),
  location VARCHAR(255),
  
  created_at TIMESTAMP DEFAULT NOW()
);

-- Time-phased resource planning (Microsoft PPM equivalent)
CREATE TABLE portfolio_resource_timephasing (
  id UUID PRIMARY KEY,
  resource_id UUID,
  program_id UUID REFERENCES programs(id),
  project_id UUID REFERENCES projects(id),
  
  -- Time periods (Microsoft PPM format)
  time_period DATE,               -- Start of week/month
  period_type VARCHAR(50),        -- daily, weekly, monthly
  
  -- Capacity
  available_capacity DECIMAL(10,2),
  allocated_capacity DECIMAL(10,2),
  remaining_capacity DECIMAL(10,2),
  
  -- Utilization
  utilization_percent DECIMAL(5,2),
  
  -- Forecasted (Microsoft PPM feature)
  forecasted_demand DECIMAL(10,2),
  
  created_at TIMESTAMP DEFAULT NOW()
);

-- Create RBS view (Microsoft PPM style)
CREATE VIEW portfolio_rbs_capacity AS
SELECT 
  rbs.rbs_code,
  rbs.rbs_name,
  rbs.department,
  SUM(rt.available_capacity) as total_capacity,
  SUM(rt.allocated_capacity) as allocated,
  SUM(rt.remaining_capacity) as available,
  AVG(rt.utilization_percent) as avg_utilization
FROM portfolio_resource_breakdown_structure rbs
LEFT JOIN portfolio_resource_timephasing rt ON rt.resource_id = rbs.id
WHERE rt.time_period >= CURRENT_DATE
  AND rt.time_period < CURRENT_DATE + INTERVAL '3 months'
GROUP BY rbs.rbs_code, rbs.rbs_name, rbs.department;
```

**ADPA Advantage**:
- ✅ AI-powered resource matching
